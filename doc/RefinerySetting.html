def value
  _value = self[:value]

 unless _value.nil?
    REPLACEMENTS.each do |current_value, new_value|
      _value = new_value if _value == current_value
    end
    _value = _value.to_i if _value.to_i.to_s == _value rescue _value
  end

  return _value
end
