def update
  @inquiry_setting.update_attributes(params[:inquiry_setting])

  if @inquiry_setting.valid?
    flash[:notice] = "'#{@inquiry_setting.name}' was successfully updated."
    redirect_to admin_inquiries_url
  else
    render :action => 'edit'
  end
end
